// 使用 Vuex 4 在 Vue.js 项目中进行状态管理的示例

// store/index.js

import { createStore } from 'vuex';

export default createStore({

state: {

count: 0,

},

mutations: {

increment(state) {

state.count++;

},

},

actions: {

incrementAsync({ commit }) {

setTimeout(() => {

commit('increment');

}, 1000);

},

},

});
